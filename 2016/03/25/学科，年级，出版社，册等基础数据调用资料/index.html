学科，年级，出版社，册等基础数据调用资料

  • 学科，年级，出版社 可以调用基础数据平台的接口 来自 MetaDataServices
  • 册 由于基础数据平台的接口是分页的 ，代码如下，可以参考sql 自己修改一下。

学科

@com.alibaba.dubbo.config.annotation.Reference
private MetaDataServices metaDataServices;

@RequestMapping(value = "findSubjectValue", method = RequestMethod.POST,produces =["application/json;charset=UTF-8"])
    public List<MetaDataVo> findSubjectValue(HttpServletRequest request){
        List<SubjectsVo> list=metaDataServices.findAllSubjects()
        List<MetaDataVo> metaDataVoList = new ArrayList<MetaDataVo>()
        for(SubjectsVo subjects: list){
                MetaDataVo metaDataVo = new MetaDataVo()
                metaDataVo.keyCode = subjects.businessKey
                metaDataVo.keyValue = subjects.name
                metaDataVoList.add(metaDataVo)
        }
        return  metaDataVoList
    }

function getSubjects(){
    $.ajax({
        url: 'metaData/findSubjectValue',
        dataType : 'json',
        type : "POST",
        success : function(data) {
            $("#subjectsId").empty()
            var vhtml="<option value=\"\">请选择</option>";
            $.each(data,function(i,item){
                vhtml+="<option value='"+item.keyCode+"'>"+item.keyValue+"</option>";
            })
            $("#subjectsId").html(vhtml);
            multipleSelect()
        }
    });
}

年级

function getGrade(){
    $("#gradeId").empty()
    $.ajax({
        url: 'metaData/findBaseGrade',
        dataType : 'text',
        type : "POST",
        success : function(data) {
            $("#gradeId").empty()
            $("#gradeId").append("<option value=\"\">请选择</option>"+data.replace('<select>','').replace('</select>',''));
        }
    });
}

@RequestMapping(value = "findBaseGrade", method = RequestMethod.POST,produces =["application/json;charset=UTF-8"])
    public String findBaseGrade(HttpServletRequest request){
        List<BaseGradeVo> list = metaDataServices.findBaseGrade()
        StringBuffer jsonStrBuf=new StringBuffer()
        jsonStrBuf.append("<select>")
        for(BaseGradeVo baseGrade:list){
            jsonStrBuf.append("<option value='"+baseGrade.businessKey +"'>"+baseGrade.name +"</option>")
        }
        jsonStrBuf.append("<select>")
        return  jsonStrBuf.toString()
    }

出版社

@com.alibaba.dubbo.config.annotation.Reference
private TextBookVersionService textBookVersionService

@RequestMapping(value = "findTextBookVersionValue", method = RequestMethod.POST,produces =["application/json;charset=UTF-8"])
    public List<MetaDataVo> findTextBookVersionValue(HttpServletRequest request){
        List<TextBookVersionVo> list = textBookVersionService.findAllTextBookVersion()
        List<MetaDataVo> metaDataVoList = new ArrayList<MetaDataVo>()
        for(TextBookVersionVo version: list){
            MetaDataVo metaDataVo = new MetaDataVo()
            metaDataVo.keyCode = version.businessKey
            metaDataVo.keyValue = version.name
            metaDataVoList.add(metaDataVo)
        }
        return  metaDataVoList
    }

册

@RequestMapping(value = 'findFasciculeByPage', method = RequestMethod.POST)
public String findFasciculeByPage(HttpServletRequest request) {
    String name = request.getParameter("name");
    String phases = request.getParameter("phases")
    String grade = request.getParameter("grade")
    String subject = request.getParameter("subject")
    String textBookVersion = request.getParameter("textBookVersion")
    String isDelete = request.getParameter("isDelete")
    Map<String,Object> queryMap = new HashMap<String,Object>()
    queryMap.put("name",name);
    queryMap.put("phases",phases);
    queryMap.put("grade",grade);
    queryMap.put("subject",subject);
    queryMap.put("textBookVersion",textBookVersion);
    queryMap.put("isDelete",isDelete)
    Map<String, Integer> currentParam = JsonHelp.parserPageParms(request)
    PageResultVo p = fasciculeServices.findFasciculeByPage(queryMap,currentParam[JsonHelp.CURRENT_SIZE], currentParam[JsonHelp.CURRENT_PAGE])
    JsonHelp.getPageJson(p, currentParam[JsonHelp.CURRENT_PAGE],currentParam[JsonHelp.CURRENT_SIZE])
}

@Override
PageResultVo findFasciculeByPage(Map<String, String> queryMap, int pageSize , int currentPage ) {
    PageResultVo pageResultVo = new PageResultVo()
    List<Fascicule> fasciculeList=new ArrayList<Fascicule>();
    List<FasciculeVo> fasciculeVoList = new ArrayList<FasciculeVo>()
    StringBuffer sqlBuf=new StringBuffer("from Fascicule a where 1=1 ");

    if(StringUtils.isNotBlank(queryMap.get("isDelete"))){
        sqlBuf.append(" and a.isDelete = '" + queryMap.get("isDelete") + "' ");
    }

    if(Const.isNotNull(queryMap.get("name"))) {
        sqlBuf.append(" and a.fascicule like '%" + queryMap.get("name")+"%' ");
    }
    if(Const.isNotNull(queryMap.get("phases"))) {
        sqlBuf.append(" and a.phasesBusinessKey ='" + queryMap.get("phases")+"'");
    }
    if(Const.isNotNull(queryMap.get("grade"))) {
        sqlBuf.append(" and a.gradeBusinessKey ='" + queryMap.get("grade")+"'");
    }
    if(Const.isNotNull(queryMap.get("subject"))) {
        sqlBuf.append(" and a.subjectBusinessKey ='" + queryMap.get("subject")+"'");
    }
    if(Const.isNotNull(queryMap.get("textBookVersion"))) {
        sqlBuf.append(" and a.textBookVersionBusinessKey ='" + queryMap.get("textBookVersion")+"'");
    }
    sqlBuf.append(" order by id desc ");
    fasciculeList = Fascicule.findAll(sqlBuf.toString(),[max:pageSize, offset: (currentPage-1)*pageSize]);
    fasciculeList.each { fascicule->
        FasciculeVo fasciculeVo=new FasciculeVo();
        ConvertSuperUtil.convertSimple(fascicule,fasciculeVo);
        if(Const.isNotNull(Phases.findByBusinessKey(fasciculeVo.phasesBusinessKey))){
            fasciculeVo.phasesName=Phases.findByBusinessKey(fasciculeVo.phasesBusinessKey).name;
        }
        if(Const.isNotNull(BaseGrade.findByBusinessKey(fasciculeVo.gradeBusinessKey))){
            fasciculeVo.gradeName=BaseGrade.findByBusinessKey(fasciculeVo.gradeBusinessKey).name;
        }
        if(Const.isNotNull(Subjects.findByBusinessKey(fasciculeVo.subjectBusinessKey))){
            fasciculeVo.subjectName=Subjects.findByBusinessKey(fasciculeVo.subjectBusinessKey).name;
        }
        if(Const.isNotNull(TextBookVersion.findByBusinessKey(fascicule.textBookVersionBusinessKey))){
            fasciculeVo.textBookVersionName=TextBookVersion.findByBusinessKey(fascicule.textBookVersionBusinessKey).name;
        }
        fasciculeVoList.add(fasciculeVo);
    }
    pageResultVo.result = fasciculeVoList
    pageResultVo.total = Fascicule.findAll(sqlBuf.toString()).size()
    return pageResultVo
}
